 441        /// <param name="licenseDecline">String used to decline a license.</param>
 442        /// <param name="licenseTextHeader">Regex which matches the line which is the start of a
 443        /// license agreement.</param>
 444        /// <param name="complete">Called when installation is complete.</param>
 445        public static void InstallPackages(
 446                string toolPath, string toolArguments,
 447                HashSet<AndroidSdkPackageNameVersion> packages,
 448                string licenseQuestion, string licenseAgree, string licenseDecline,
 449                Regex licenseTextHeader, Action<bool> complete) {
 450            PlayServicesResolver.Log(String.Format("Install Android SDK packages\n" +
 451                                                   "\n" +
 452                                                   "{0} {1}\n",
 453                                                   toolPath, toolArguments),
 454                                     level: LogLevel.Verbose);
 455            // Display the license retrieval dialog.
 456            DisplayInstallLicenseDialog(
 457                toolPath, toolArguments, true,
 458                new LicenseResponder(licenseQuestion, licenseDecline), packages,
 459                (CommandLine.Result licensesResult) => {
 460                    if (licensesResult.exitCode != 0) {
 461                        complete(false);
 462                        return;
 463                    }
 464                    // Get the license text.
 465                    var licensesLines = new List<string>();
 466                    bool foundLicenses = false;
 467                    foreach (var line in CommandLine.SplitLines(licensesResult.stdout)) {
 468                        foundLicenses = foundLicenses || licenseTextHeader.Match(line).Success;
 469                        if (foundLicenses) licensesLines.Add(line);
 470                    }
 471                    if (licensesLines.Count == 0) {
 472                        LogInstallLicenseResult(toolPath, toolArguments, false, packages,
 473                                                licensesResult);
 474                        complete(true);
 475                        return;
 476                    }
 477                    // Display the license agreement dialog.
 478                    DisplayLicensesDialog(
 479                        String.Join("\n", licensesLines.ToArray()),
 480                        (bool agreed) => {
 481                            if (!agreed) {
 482                                complete(false);
 483                                return;
 484                            }
 485                            // Finally install the packages agreeing to the license questions.
 486                            DisplayInstallLicenseDialog(
 487                                toolPath, toolArguments, false,
 488                                new LicenseResponder(licenseQuestion, licenseAgree), packages,
 489                                (CommandLine.Result installResult) => {
 490                                    complete(installResult.exitCode == 0);
 491                                });
 492                        });
 493                });
 494        }

 625        /// </summary>

 668        /// <summary>
 669        /// Parse "android list sdk -u -e -a" output.
 670        /// </summary>
 671        private AndroidSdkPackageCollection ParseAndroidListSdkOutput(
 672                string androidListSdkOutput) {
 673            var packages = new AndroidSdkPackageCollection();
 674            AndroidSdkPackage currentPackage = null;
 675            foreach (string line in CommandLine.SplitLines(androidListSdkOutput)) {
 676                // Check for the start of a new package entry.
 677                if (line.StartsWith("---")) {
 678                    currentPackage = null;
 679                    continue;
 680                }
 681                Match match;
 682                // If this is the start of a package description, add a package.
 683                match = PACKAGE_ID_REGEX.Match(line);
 684                if (match.Success) {
 685                    // TODO(smiles): Convert the legacy package name to a new package name.
 686                    currentPackage = new AndroidSdkPackage { LegacyName = match.Groups[1].Value };
 687                    packages[currentPackage.Name].Add(currentPackage);
 688                    continue;
 689                }
 690                if (currentPackage == null) continue;
 691
 692                // Add a package description.
 693                match = PACKAGE_DESCRIPTION_REGEX.Match(line);
 694                if (match.Success) {
 695                    currentPackage.Description = match.Groups[1].Value;
 696                    continue;
 697                }
 698                // Parse the install path and record whether the package is installed.
 699                match = PACKAGE_INSTALL_LOCATION_REGEX.Match(line);
 700                if (match.Success) {
 701                    currentPackage.Installed = File.Exists(
 702                        Path.Combine(Path.Combine(sdkPath, match.Groups[1].Value),
 703                                     "source.properties"));
 704                }
 705            }
 706            return packages;
 707        }

 739    /// <summary>
 740    /// Interacts with the Android SDK manager "sdkmanager".
 741    /// </summary>
 742    internal class SdkManager : IAndroidSdkManager {
 743        /// Name of the SDK manager command line tool.
 744        public const string TOOL_NAME = "sdkmanager";
 745
 746        // Marker followed by the list of installed packages.
 747        private const string INSTALLED_PACKAGES_HEADER = "installed packages:";
 748        // Marker followed by the list of available packages.
 749        private const string AVAILABLE_PACKAGES_HEADER = "available packages:";
 750        // Marker followed by the list of available package updates.
 751        private const string AVAILABLE_UPDATES_HEADER = "available updates:";
 752
 753        /// Minimum version of the package that supports the --verbose flag.
 754        public static long MINIMUM_VERSION_FOR_VERBOSE_OUTPUT =
 755            AndroidSdkPackageNameVersion.ConvertVersionStringToInteger("26.0.2");
 756
 757        // Path to the SDK manager tool.
 758        private string toolPath;
 759        // Metadata for this package.
 760        private AndroidSdkPackage toolsPackage;
 761
 762        /// <summary>
 763        /// Initialize this instance.
 764        /// </summary>
 765        /// <param name="toolPath">Path of the android tool.</param>
 766        public SdkManager(string toolPath) {
 767            this.toolPath = toolPath;
 768            var toolsDir = Path.GetDirectoryName(Path.GetDirectoryName(toolPath));
 769            var sdkDir = Path.GetDirectoryName(toolsDir);
 770            toolsPackage = AndroidSdkPackage.ReadFromSourceProperties(
 771                sdkDir, toolsDir.Substring((sdkDir + Path.PathSeparator).Length));
 772        }
 773
 774        /// <summary>
 775        /// Read the metadata for the package that contains the package manager.
 776        /// </summary>
 777        public AndroidSdkPackage Package { get { return toolsPackage; } }
 778
 779        /// <summary>
 780        /// Parse "sdkmanager --list --verbose" output.
 781        /// NOTE: The --verbose output format is only reported by sdkmanager 26.0.2 and above.
 782        /// </summary>
 783        private AndroidSdkPackageCollection ParseListVerboseOutput(
 784                string sdkManagerListVerboseOutput) {
 785            var packages = new AndroidSdkPackageCollection();
 786            // Whether we're parsing a set of packages.
 787            bool parsingPackages = false;
 788            // Whether we're parsing within the set of installed packages vs. available packages.
 789            bool parsingInstalledPackages = false;
 790            // Fields of the package being parsed.
 791            AndroidSdkPackage currentPackage = null;
 792            foreach (var rawLine in CommandLine.SplitLines(sdkManagerListVerboseOutput)) {
 793                var line = rawLine.Trim();
 794                var lowerCaseLine = line.ToLower();
 795                if (lowerCaseLine == AVAILABLE_UPDATES_HEADER) {
 796                    parsingPackages = false;
 797                    continue;
 798                }
 799                bool installedPackagesLine = lowerCaseLine == INSTALLED_PACKAGES_HEADER;
 800                bool availablePackagesLine = lowerCaseLine == AVAILABLE_PACKAGES_HEADER;
 801                if (installedPackagesLine || availablePackagesLine) {
 802                    parsingPackages = true;
 803                    parsingInstalledPackages = installedPackagesLine;
 804                    continue;
 805                } else if (line.StartsWith("---")) {
 806                    // Ignore section separators.
 807                    continue;
 808                } else if (String.IsNullOrEmpty(line)) {
 809                    if (currentPackage != null &&
 810                        !(String.IsNullOrEmpty(currentPackage.Name) ||
 811                          String.IsNullOrEmpty(currentPackage.VersionString))) {
 812                        packages[currentPackage.Name].Add(currentPackage);
 813                    }
 814                    currentPackage = null;
 815                    continue;
 816                } else if (!parsingPackages) {
 817                    continue;
 818                }
 819                // Fields of the package are indented.
 820                bool indentedLine = rawLine.StartsWith("    ");
 821                if (!indentedLine) {
 822                    // If this isn't an indented line it should be a package name.
 823                    if (currentPackage == null) {
 824                        currentPackage = new AndroidSdkPackage {
 825                            Name = line,
 826                            Installed = parsingInstalledPackages
 827                        };
 828                    }
 829                } else if (currentPackage != null) {
 830                    // Parse the package field.
 831                    var fieldSeparatorIndex = line.IndexOf(":");
 832                    if (fieldSeparatorIndex >= 0) {
 833                        var fieldName = line.Substring(0, fieldSeparatorIndex).Trim().ToLower();
 834                        var fieldValue = line.Substring(fieldSeparatorIndex + 1).Trim();
 835                        if (fieldName == "description") {
 836                            currentPackage.Description = fieldValue;
 837                        } else if (fieldName == "version") {
 838                            currentPackage.VersionString = fieldValue;
 839                        }
 840                    }
 841                }
 842            }
 843            return packages;
 844        }
 845
 846        /// <summary>
 847        /// Parse "sdkmanager --list" output.
 848        /// </summary>
 849        /// <returns>Dictionary of packages bucketed by package name</returns>
 850        private AndroidSdkPackageCollection ParseListOutput(
 851                string sdkManagerListOutput) {
 852            var packages = new AndroidSdkPackageCollection();
 853            // Whether we're parsing a set of packages.
 854            bool parsingPackages = false;
 855            // Whether we're parsing within the set of installed packages vs. available packages.
 856            bool parsingInstalledPackages = false;
 857            // Whether we're parsing the contents of the package table vs. the header.
 858            bool inPackageTable = false;
 859            foreach (var rawLine in CommandLine.SplitLines(sdkManagerListOutput)) {
 860                var line = rawLine.Trim();
 861                var lowerCaseLine = line.ToLower();
