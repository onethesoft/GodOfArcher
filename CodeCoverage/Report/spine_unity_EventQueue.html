 419      this.events.Clear(false);
 420      from.nextAnimationLast = animationTime;
 421      from.nextTrackLast = from.trackTime;
 422
 423      return mix;

 496    /// <summary>
 497    /// Applies the rotate timeline, mixing with the current pose while keeping the same rotation direction chosen as th
 498    /// the first time the mixing was applied.</summary>
 499    static private void ApplyRotateTimeline (RotateTimeline timeline, Skeleton skeleton, float time, float alpha, MixBle
 500      float[] timelinesRotation, int i, bool firstFrame) {
 501
 502      if (firstFrame) timelinesRotation[i] = 0;
 503
 504      if (alpha == 1) {
 505        timeline.Apply(skeleton, 0, time, null, 1, blend, MixDirection.In);
 506        return;
 507      }
 508
 509      Bone bone = skeleton.bones.Items[timeline.boneIndex];
 510      if (!bone.active) return;
 511
 512      float[] frames = timeline.frames;
 513      float r1, r2;
 514      if (time < frames[0]) { // Time is before first frame.
 515        switch (blend) {
 516          case MixBlend.Setup:
 517            bone.rotation = bone.data.rotation;
 518            return;
 519          default:
 520            return;
 521          case MixBlend.First:
 522            r1 = bone.rotation;
 523            r2 = bone.data.rotation;
 524            break;
 525        }
 526      } else {
 527        r1 = blend == MixBlend.Setup ? bone.data.rotation : bone.rotation;
 528        if (time >= frames[frames.Length - RotateTimeline.ENTRIES]) // Time is after last frame.
 529          r2 = bone.data.rotation + frames[frames.Length + RotateTimeline.PREV_ROTATION];
 530        else {
 531          // Interpolate between the previous frame and the current frame.
 532          int frame = Animation.BinarySearch(frames, time, RotateTimeline.ENTRIES);
 533          float prevRotation = frames[frame + RotateTimeline.PREV_ROTATION];
 534          float frameTime = frames[frame];
 535          float percent = timeline.GetCurvePercent((frame >> 1) - 1,
 536            1 - (time - frameTime) / (frames[frame + RotateTimeline.PREV_TIME] - frameTime));
 537
 538          r2 = frames[frame + RotateTimeline.ROTATION] - prevRotation;
 539          r2 -= (16384 - (int)(16384.499999999996 - r2 / 360)) * 360;
 540          r2 = prevRotation + r2 * percent + bone.data.rotation;
 541          r2 -= (16384 - (int)(16384.499999999996 - r2 / 360)) * 360;
 542        }
 543      }
 544
 545      // Mix between rotations using the direction of the shortest route on the first frame.
 546      float total, diff = r2 - r1;
 547      diff -= (16384 - (int)(16384.499999999996 - diff / 360)) * 360;
 548      if (diff == 0) {
 549        total = timelinesRotation[i];
 550      } else {
 551        float lastTotal, lastDiff;
 552        if (firstFrame) {
 553          lastTotal = 0;
 554          lastDiff = diff;
 555        } else {
 556          lastTotal = timelinesRotation[i]; // Angle and direction of mix, including loops.
 557          lastDiff = timelinesRotation[i + 1]; // Difference between bones.
 558        }
 559        bool current = diff > 0, dir = lastTotal >= 0;
 560        // Detect cross at 0 (not 180).
 561        if (Math.Sign(lastDiff) != Math.Sign(diff) && Math.Abs(lastDiff) <= 90) {
 562          // A cross after a 360 rotation is a loop.
 563          if (Math.Abs(lastTotal) > 180) lastTotal += 360 * Math.Sign(lastTotal);
 564          dir = current;
 565        }
 566        total = diff + lastTotal - lastTotal % 360; // Store loops as part of lastTotal.
 567        if (dir != current) total += 360 * Math.Sign(lastTotal);
 568        timelinesRotation[i] = total;
 569      }
 570      timelinesRotation[i + 1] = diff;
 571      r1 += total * alpha;
 572      bone.rotation = r1 - (16384 - (int)(16384.499999999996 - r1 / 360)) * 360;
 573    }

 575    private void QueueEvents (TrackEntry entry, float animationTime) {
 576      float animationStart = entry.animationStart, animationEnd = entry.animationEnd;
 577      float duration = animationEnd - animationStart;
 578      float trackLastWrapped = entry.trackLast % duration;
 579
 580      // Queue events before complete.
 581      var events = this.events;
 582      var eventsItems = events.Items;
 583      int i = 0, n = events.Count;
 584      for (; i < n; i++) {
 585        Event e = eventsItems[i];
 586        if (e.time < trackLastWrapped) break;
 587        if (e.time > animationEnd) continue; // Discard events outside animation start/end.
 588        queue.Event(entry, e);
 589      }
 590
 591      // Queue complete if completed a loop iteration or the animation.
 592      bool complete = false;
 593      if (entry.loop)
 594        complete = duration == 0 || (trackLastWrapped > entry.trackTime % duration);
 595      else
 596        complete = animationTime >= animationEnd && entry.animationLast < animationEnd;
 597      if (complete) queue.Complete(entry);
 598
 599      // Queue events after complete.
 600      for (; i < n; i++) {
 601        Event e = eventsItems[i];
 602        if (e.time < animationStart) continue; // Discard events outside animation start/end.
 603        queue.Event(entry, eventsItems[i]);
 604      }
 605    }

 733      TrackEntry last = ExpandToIndex(trackIndex);
 734      if (last != null) {
